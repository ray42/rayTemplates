%% Maths vectors, these are bold.
\newcommand{\mathvec}[1]{\bm{#1}}
\newcommand{\unitvec}[1]{\hat{#1}}


% Symbols used, this is the basic symbol, should not be used like this.
\newcommand{\normalsym}{n} % Normal vector is lower case n
\newcommand{\timesym}{t} % Time
\newcommand{\tangentsym}{t} % Tangent, same as time.
\newcommand{\fluidtractionsym}{\uptau} % Needs upgreek package
\newcommand{\stresstensorsym}{\tau}


% Do the tangent component/vector requires package xifthen
\newcommand{\rrrunitnorm}[1][]{%
  \ifthenelse{\isempty{#1}}%
    {\unitvec{\mathvec{\normalsym}}}% if #1 is empty, is a vector, bold it.
    {\unitvec{\normalsym}_{#1}}% if #1 is not empty, is component form.
}

% Do the time component/vector requires package xifthen
\newcommand{\rrrtime}[1][]{%
  \ifthenelse{\isempty{#1}}%
    {\timesym}% if #1 is empty, is just t
    {\timesym^{#1}}% if #1 is not empty, is component
}

% Do the tangent component/vector requires package xifthen
\newcommand{\rrrtang}[1][]{%
  \ifthenelse{\isempty{#1}}%
    {\mathvec{\tangentsym}}% if #1 is empty, is just t
    {\tangentsym_{#1}}% if #1 is not empty, is component
}

% Do the tangent component/vector requires package xifthen
\newcommand{\rrrunittang}[1][]{%
  \ifthenelse{\isempty{#1}}%
    {\unitvec{\mathvec{\tangentsym}}}% if #1 is empty, is just t
    {\unitvec{\tangentsym}_{#1}}% if #1 is not empty, is component
}

% Do the traction component/vector requires package xifthen
\newcommand{\rrrftraction}[1][]{%
  \ifthenelse{\isempty{#1}}%
    {\mathvec{\fluidtractionsym}}% if #1 is empty, is just t
    {\fluidtractionsym_{#1}}% if #1 is not empty, is component
}

% Do the traction component/vector requires package xifthen
\newcommand{\rrrstresstensor}[1][]{%
  \ifthenelse{\isempty{#1}}%
    {\mathvec{\stresstensorsym}}% if #1 is empty, is just t
    {\stresstensorsym_{#1}}% if #1 is not empty, is component
}
